import binascii

hex To String:
	binascii.a2b_hex("d5c5c8fd")
	binascii.a2b_hex("d5c5c8fd").decode('gbk')
	binascii.a2b_hex("e4bda0e5a5bde5958a").decode('utf-8')


String To Hex:
	b = u"你好啊"   		# must be unicode
	a = binascii.b2a_hex(b.encode('utf-8'))   # utf-8 hex
	a = binascii.b2a_hex(b.encode('gbk'))     # gbk hex
	a = binascii.b2a_hex(b.encode('gb2312'))  # gb2312 hex

	b = "你好啊"  
	a = binascii.b2a_hex(b)   # utf-8 hex
	a = binascii.b2a_hex(b.decode('utf-8').encode('gbk'))
